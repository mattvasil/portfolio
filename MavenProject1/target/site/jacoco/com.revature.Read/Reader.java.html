package com.revature.Read;
import com.revature.util.ConnectionFactory;
import org.apache.log4j.Logger;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.util.ArrayList;

public class Reader {
    Logger logger = Logger.getLogger(Reader.class);

    public static void readColumnWhere(String tableName, String columnName, String condition){

        String read = "select "+columnName+" from "+tableName+" where "+condition;
        ArrayList<Object> table = new ArrayList<>();
        try {
            PreparedStatement ps = ConnectionFactory.getConnection().prepareStatement(read);
            ResultSet rs = ps.executeQuery();

            while(rs.next()){
                table.add(rs.getObject(columnName));
            }

        } catch (SQLException e) {
            e.printStackTrace();
        }
        System.out.println(table);
    }

    public static void readRecordWhere(String tableName, String condition){
        String read = "select * from "+tableName+" where "+condition;

        try {
            PreparedStatement ps = ConnectionFactory.getConnection().prepareStatement(read);
            ResultSet rs = ps.executeQuery();

            while(rs.next()){
                for(int i = 1; i <= rs.getMetaData().getColumnCount(); i++){
                    System.out.println(rs.getObject(i));
                }
            }

        } catch (SQLException e) {
            e.printStackTrace();
        }

    }

    public static void readColumn(String tableName, String columnName){
        String read = "select "+columnName+" from "+tableName;
        try {
            PreparedStatement ps = ConnectionFactory.getConnection().prepareStatement(read);
            ResultSet rs = ps.executeQuery();

            while (rs.next()){
                System.out.println(rs.getObject(columnName));
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }

}
